## Quick start


        </StepL>

                    },

                        code: `npm run dev`
                    }
                ]}
            </CodeTabs>
        </StepR>

## Installation

<StepSection>
    <Step $row>
        <StepL>
            ### <StepNum />Create a project
            If you don't have a Laravel project, create one first. It's recommended to refer to [Installation - Laravel](https://laravel.com/docs/10.x)
        </StepL>
        <StepR>
            ```bash name=Terminal
            composer create-project laravel/laravel __example-app__
            ```
        </StepR>
    </Step>
    <Step $row>
        <StepL>
            ### <StepNum />Initialize configuration file
            Run `npm create @master/css@rc` to create a configuration file [master.css.js](/docs/configuration).
        </StepL>
        <StepR>
            ```bash name=Terminal
            npm create @master/css@rc
            ```
        </StepR>
    </Step>
    <Step $row>
        <StepL>
            ### <StepNum />Install Master CSS
            Install Master CSS into your project via package managers.
        </StepL>
        <StepR>
            ```install
            **@master/css-runtime@rc**
            ```
        </StepR>
    </Step>
    <Step $row>
        <StepL>
            ### <StepNum />Set up CSS runtime engine
            Import `master.css.js` config and initialize the runtime engine in your `resources/js/app.js` file.
        </StepL>
        <StepR>
            ```js name=resources/js/app.js
            import './bootstrap'
            + import { initCSSRuntime } from '@master/css-runtime'
            + import config from 'master.css'

            + initCSSRuntime(config)
            ```
        </StepR>
    </Step>
    <Step $row>
        <StepL>
            ### <StepNum />Set up your view layout
            Include the following in your `resources/views/layouts/welcome.blade.php`:
            - Import the entry by adding `@vite('resources/js/app.js')`
            - Add `css display: none` to `html <html>` to [avoid FOUC](/docs/flash-of-unstyled-content)
        </StepL>
        <StepR>
            ```html name=resources/views/layouts/welcome.blade.php
            <!DOCTYPE html>
            <html style="**display: none**" lang="{{ str_replace('_', '-', app()->getLocale()) }}">
                <head>
                    …
            +        @vite('resources/js/app.js')
                </head>
                …
            </html>
            ```
        </StepR>
    </Step>
    <Step $row>
        <StepL>
            ### <StepNum />Launch server
            Run Laravel and Vite development servers by splitting two terminals.
        </StepL>
        <StepR>
            <CodeTabs>
                {[
                    {
                        name: 'PHP',
                        lang: 'bash',
                        code: `php artisan serve`
                    },
                    {
                        name: 'Vite',
                        lang: 'bash',
                        code: `npm run dev`
                    }
                ]}
            </CodeTabs>
        </StepR>
    </Step>
    <Step>
        ### <StepNum />Hello world!
        Now style your first element using Master CSS syntax!
        ```html name=resources/views/layouts/welcome.blade.php
        …
        <body>
            <h1 class="**font:40** **font:heavy** **italic** **m:12x** **text:center**">
                Hello World
            </h1>
        </body>
        …
        ```
        <p className="mt:12x">Open your browser to watch the changes.</p>
        <HelloWorld url="127.0.0.1:8000" />
    </Step>
</StepSection>
